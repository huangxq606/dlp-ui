#!/bin/sh #基于svn的项目管理初始化操作 #变量定义 if [ $# -ne 2 ] then echo "============generate tag of app=============" echo "Usage: ./pre_svn.sh scm workspace" exit 1 fi #项目svn路径 scm=$1 #项目工作空间 workspace=$2 if [ ! -d $workspace ]; then svn checkout $scm $workspace fi cd $workspace svn revert -R $workspace #清除未受版本控制的文件 svn st|grep '^?'|awk '{print $2}'|xargs rm -rf
